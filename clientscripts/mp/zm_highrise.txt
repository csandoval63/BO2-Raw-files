// 0x1D58
main()
{
	level thread clientscripts\mp\zm_highrise_ffotd::main_start();
	level.default_start_location = "rooftop";
	level.default_game_mode = "zclassic";
	level._no_water_risers = 1;
	level.setupcustomcharacterexerts = ::setup_personality_character_exerts;
	level.zombiemode_using_doubletap_perk = 1;
	level.zombiemode_using_juggernaut_perk = 1;
	level.zombiemode_using_revive_perk = 1;
	level.zombiemode_using_sleightofhand_perk = 1;
	level.zombiemode_using_chugabud_perk = 1;
	level.zombiemode_using_additionalprimaryweapon_perk = 1;
	clientscripts\mp\zm_highrise_classic::init_escape_elevators_animtree();
	clientscripts\mp\zm_highrise_classic::init_perk_elvators_animtree();
	start_zombie_stuff();
	init_gamemodes();
	register_client_fields();
	clientscripts\mp\zm_highrise_fx::main();
	init_level_specific_wall_buy_fx();
	level._override_eye_fx = level._effect["blue_eyes"];
	thread clientscripts\mp\zm_highrise_amb::main();
	zombe_gametype_premain();
	claymores = getstructarray( "claymore_purchase", "targetname" );
	foreach ( struct in claymores )
	{
		weapon_model = getstruct( struct.target, "targetname" );
		weapon_model.script_vector = vector_scale( ( 0, -1, 0 ), 90 );
	}
	level thread clientscripts\mp\zm_highrise_ffotd::main_end();
	setsaveddvar( "r_skyTransition", 1 );
	clientscripts\mp\zm_highrise_fx::setup_prop_anims();
	level thread clientscripts\mp\zombies\_zm_perks::chugabud_setup_afterlife_filters();
	waitforclient( 0 );
	level thread highrise_watch_for_power();
	level thread init_fog_vol_to_visionset();
	level thread clientscripts\mp\zm_highrise_sq::init();
// SP = 0x0 - check OK
}

// 0x24BC
highrise_watch_for_power()
{
	return;
	looped_once = 0;
	level.power_on = 0;
	level notify( "power_controlled_light" );
	fog_vol_to_visionset_set_suffix( "_off" );
	fog_vol_to_visionset_set_info( 0, "zm_highrise" );
	fog_vol_to_visionset_set_info( 1, "zm_highrise_ext" );
	level waittill_any( "power_on", "pwr", "ZPO" );
	level notify( "power_controlled_light" );
	clientscripts\mp\_fx::activate_exploder( 10 );
	fog_vol_to_visionset_set_suffix( "_on" );
	players = getlocalplayers();
	i = 0;
	level.current_fog = 2;
	setworldfogactivebank( i, level.current_fog );
	i++;
	level waittill_any( "pwo", "ZPOff" );
	looped_once = 1;
// SP = 0x0 - check OK
}
